## 1. 后端架构概述

### 1.1 技术栈
- **框架**: FastAPI 0.110+
- **语言**: Python 3.11+
- **数据库**: PostgreSQL 15+ (主库) + Redis 7.x (缓存)
- **ORM**: SQLAlchemy 2.x
- **迁移**: Alembic 1.x
- **认证**: JWT (python-jose)
- **密码**: Passlib (bcrypt)
- **验证**: Pydantic 2.x
- **异步**: asyncio + asyncpg

### 1.2 分层架构

```
┌─────────────────────────────────────┐
│         API Layer (路由层)           │
│    处理HTTP请求、参数验证、响应格式    │
├─────────────────────────────────────┤
│       Service Layer (业务层)         │

│        CRUD Layer (数据访问层)        │

├─────────────────────────────────────┤
│       Model Layer (模型层)           │

```

```python
# app/crud/user.py
from sqlalchemy.orm import Session
from app.models.user import User
from app.schemas.user import UserRegister, UserUpdate
from app.core.security import get_password_hash

class CRUDUser:
    def get_by_id(self, db: Session, user_id: int) -> Optional[User]:
        return db.query(User).filter(User.id == user_id).first()
    
    def get_by_username(self, db: Session, username: str) -> Optional[User]:
        return db.query(User).filter(User.username == username).first()
    
    def get_by_phone(self, db: Session, phone: str) -> Optional[User]:
        return db.query(User).filter(User.phone == phone).first()
    
    def create(self, db: Session, user_in: UserRegister) -> User:
        db_user = User(
            username=user_in.username,
            phone=user_in.phone,
            email=user_in.email,
            password_hash=get_password_hash(user_in.password),
            nickname=user_in.username
        )
        db.add(db_user)
        db.commit()
        db.refresh(db_user)
        return db_user
    
    def update(self, db: Session, user_id: int, user_in: UserUpdate) -> User:
        db_user = self.get_by_id(db, user_id)
        update_data = user_in.dict(exclude_unset=True)
        for field, value in update_data.items():
            setattr(db_user, field, value)
        db.commit()
        db.refresh(db_user)
        return db_user
```

```python
# app/services/user_service.py
from sqlalchemy.orm import Session
from app.crud.user import crud_user
from app.schemas.user import UserRegister
from app.core.exceptions import BusinessException
from app.utils.sms import send_sms_code, verify_sms_code

class UserService:
    def register(self, db: Session, user_in: UserRegister) -> User:
        # 1. 验证验证码
        if not verify_sms_code(user_in.phone, user_in.code):
            raise BusinessException("验证码错误或已过期")
        
        # 2. 检查用户名是否存在
        if crud_user.get_by_username(db, user_in.username):
            raise BusinessException("用户名已存在")
        
        # 3. 检查手机号是否存在
        if crud_user.get_by_phone(db, user_in.phone):
            raise BusinessException("手机号已注册")
        
        # 4. 创建用户
        user = crud_user.create(db, user_in)
        return user
    
    def send_register_code(self, phone: str) -> bool:
        # 1. 检查手机号是否已注册
        # 2. 生成验证码
        # 3. 发送短信
        # 4. 存储到Redis
        return send_sms_code(phone, "register")
```

### 4.3 业务逻辑

```python
# app/services/product_service.py
from sqlalchemy.orm import Session
from app.models.product import Product, ProductSKU, ProductStatus
from app.schemas.product import ProductCreate, ProductUpdate

class ProductService:
    def create_product(self, db: Session, merchant_id: int, product_in: ProductCreate) -> Product:
        # 1. 创建商品
        db_product = Product(
            merchant_id=merchant_id,
            category_id=product_in.category_id,
            name=product_in.name,
            description=product_in.description,
            images=product_in.images,
            video=product_in.video,
            status=ProductStatus.PENDING
        )
        db.add(db_product)
        db.flush()
        
        # 2. 创建SKU
        for sku_in in product_in.skus:
            db_sku = ProductSKU(
                product_id=db_product.id,
                **sku_in.dict()
            )
            db.add(db_sku)
        
        db.commit()
        db.refresh(db_product)
        return db_product
    
    def update_product(self, db: Session, product_id: int, product_in: ProductUpdate) -> Product:
        db_product = db.query(Product).filter(Product.id == product_id).first()
        if not db_product:
            raise BusinessException("商品不存在")
        
        # 更新商品基本信息
        update_data = product_in.dict(exclude_unset=True, exclude={"skus"})
        for field, value in update_data.items():
            setattr(db_product, field, value)
        
        # 更新SKU(简化处理:删除旧的,创建新的)
        if product_in.skus is not None:
            db.query(ProductSKU).filter(ProductSKU.product_id == product_id).delete()
            for sku_in in product_in.skus:
                db_sku = ProductSKU(product_id=product_id, **sku_in.dict())
                db.add(db_sku)
        
        db.commit()
        db.refresh(db_product)
        return db_product
    
    def get_product_list(
        self, 
        db: Session, 
        category_id: Optional[int] = None,
        keyword: Optional[str] = None,
        min_price: Optional[Decimal] = None,
        max_price: Optional[Decimal] = None,
        sort_by: str = "default",
        page: int = 1,
        page_size: int = 20
    ):
        query = db.query(Product).filter(Product.status == ProductStatus.APPROVED)
        
        # 分类筛选
        if category_id:
            query = query.filter(Product.category_id == category_id)
        
        # 关键词搜索
        if keyword:
            query = query.filter(Product.name.ilike(f"%{keyword}%"))
        
        # 价格筛选(需要JOIN SKU表)
        if min_price or max_price:
            query = query.join(ProductSKU)
            if min_price:
                query = query.filter(ProductSKU.price >= min_price)
            if max_price:
                query = query.filter(ProductSKU.price <= max_price)
        
        # 排序
        if sort_by == "price_asc":
            query = query.join(ProductSKU).order_by(ProductSKU.price.asc())
        elif sort_by == "price_desc":
            query = query.join(ProductSKU).order_by(ProductSKU.price.desc())
        elif sort_by == "sales":
            query = query.order_by(Product.sales.desc())
        elif sort_by == "new":
            query = query.order_by(Product.created_at.desc())
        
        # 分页
        total = query.count()
        products = query.offset((page - 1) * page_size).limit(page_size).all()
        
        return {
            "total": total,
            "page": page,
            "page_size": page_size,
            "items": products
        }

product_service = ProductService()
```

```python
import string

class OrderService:
    def create_order(
        self,

        user_id: int,
        items: List[dict],  # [{"sku_id": 1, "quantity": 2}]
        address_id: int,
        remark: Optional[str] = None
    ) -> Order:
        # 1. 获取收货地址
        address = db.query(Address).filter(Address.id == address_id).first()
        if not address:
            raise BusinessException("收货地址不存在")
        
        # 2. 计算订单金额,验证库存
        total_amount = Decimal(0)
        order_items = []
        merchant_id = None
        
        for item in items:
            sku = db.query(ProductSKU).filter(ProductSKU.id == item["sku_id"]).first()
            if not sku:
                raise BusinessException(f"SKU {item['sku_id']} 不存在")
            
            if sku.stock < item["quantity"]:
                raise BusinessException(f"{sku.product.name} 库存不足")
            
            # 验证商品属于同一商家
            if merchant_id is None:
                merchant_id = sku.product.merchant_id
            elif merchant_id != sku.product.merchant_id:
                raise BusinessException("不支持跨店铺下单")
            
            subtotal = sku.price * item["quantity"]
            total_amount += subtotal
            
            order_items.append({
                "product_id": sku.product_id,
                "sku_id": sku.id,
                "product_name": sku.product.name,
                "sku_name": sku.sku_name,
                "image": sku.image or sku.product.images[0],
                "price": sku.price,
                "quantity": item["quantity"],
                "subtotal": subtotal
            })
        
        # 3. 计算运费(简化处理,固定运费)
        freight = Decimal(10)
        
        # 4. 计算实付金额
        actual_amount = total_amount + freight
        
        # 5. 生成订单号
        order_no = self.generate_order_no()
        
        # 6. 创建订单
        db_order = Order(
            order_no=order_no,
            user_id=user_id,
            merchant_id=merchant_id,
            receiver_name=address.receiver_name,
            receiver_phone=address.receiver_phone,
            receiver_address=f"{address.province}{address.city}{address.district}{address.detail}",
            total_amount=total_amount,
            freight=freight,
            discount_amount=Decimal(0),
            actual_amount=actual_amount,
            remark=remark,
            status=OrderStatus.PENDING_PAYMENT
        )
        db.add(db_order)
        db.flush()
        
        # 7. 创建订单商品
        for item_data in order_items:
            db_item = OrderItem(order_id=db_order.id, **item_data)
            db.add(db_item)
        
        # 8. 锁定库存
        for item in items:
            db.query(ProductSKU).filter(ProductSKU.id == item["sku_id"]).update({
                "stock": ProductSKU.stock - item["quantity"]
            })
        
        db.commit()
        db.refresh(db_order)
        return db_order
    
    def generate_order_no(self) -> str:
        """生成订单号: 时间戳 + 随机字符"""
        import time
        timestamp = str(int(time.time()))
        random_str = ''.join(random.choices(string.digits, k=6))
        return timestamp + random_str
    
    def pay_order(self, db: Session, order_id: int) -> Order:
        """支付订单(MVP阶段模拟支付)"""
        db_order = db.query(Order).filter(Order.id == order_id).first()
        if not db_order:
            raise BusinessException("订单不存在")
        
        if db_order.status != OrderStatus.PENDING_PAYMENT:
            raise BusinessException("订单状态不正确")
        
        # 更新订单状态
        db_order.status = OrderStatus.PENDING_SHIPMENT
        db_order.paid_at = datetime.now()
        
        db.commit()
        db.refresh(db_order)
        return db_order
    
    def ship_order(
        self,
        db: Session,
        order_id: int,
        logistics_company: str,
        logistics_no: str
    ) -> Order:
        """发货"""
        db_order = db.query(Order).filter(Order.id == order_id).first()
        if not db_order:
            raise BusinessException("订单不存在")
        
        if db_order.status != OrderStatus.PENDING_SHIPMENT:
            raise BusinessException("订单状态不正确")
        
        db_order.status = OrderStatus.SHIPPED
        db_order.logistics_company = logistics_company
        db_order.logistics_no = logistics_no
        db_order.shipped_at = datetime.now()
        
        db.commit()
        db.refresh(db_order)
        return db_order
    
    def confirm_order(self, db: Session, order_id: int) -> Order:
        """确认收货"""
        db_order = db.query(Order).filter(Order.id == order_id).first()
        if not db_order:
            raise BusinessException("订单不存在")
        
        if db_order.status != OrderStatus.SHIPPED:
            raise BusinessException("订单状态不正确")
        
        db_order.status = OrderStatus.COMPLETED
        db_order.completed_at = datetime.now()
        
        # 增加商品销量
        for item in db_order.items:
            db.query(Product).filter(Product.id == item.product_id).update({
                "sales": Product.sales + item.quantity
            })
        
        db.commit()
        db.refresh(db_order)
        return db_order

order_service = OrderService()
```

```python
# app/core/security.py
from datetime import datetime, timedelta
from jose import jwt
from passlib.context import CryptContext
from app.core.config import settings

pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

def create_access_token(subject: int, role: str, expires_delta: timedelta = None) -> str:
    if expires_delta:
        expire = datetime.utcnow() + expires_delta
    else:
        expire = datetime.utcnow() + timedelta(hours=2)
    
    to_encode = {
        "sub": str(subject),
        "role": role,
        "exp": expire
    }
    encoded_jwt = jwt.encode(to_encode, settings.SECRET_KEY, algorithm="HS256")
    return encoded_jwt

def verify_password(plain_password: str, hashed_password: str) -> bool:
    return pwd_context.verify(plain_password, hashed_password)

def get_password_hash(password: str) -> str:
    return pwd_context.hash(password)
```
