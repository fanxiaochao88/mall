```typescript
import { useState, useMemo } from 'react'
import type { ProductSKU } from '@/types/product'

/**
 * SKU 选择器逻辑封装
 */
export function useSKUSelector(skus: ProductSKU[]) {
  const [selectedAttributes, setSelectedAttributes] = useState<Record<string, string>>({})

  // 提取所有属性
  const attributes = useMemo(() => {
    const attrs: Record<string, Set<string>> = {}

    skus.forEach(sku => {
      if (sku.attributes) {
        Object.entries(sku.attributes).forEach(([key, value]) => {
          if (!attrs[key]) attrs[key] = new Set()
          attrs[key].add(value)
        })
      }
    })

    // 转换为数组
    return Object.entries(attrs).reduce((acc, [key, values]) => {
      acc[key] = Array.from(values)
      return acc
    }, {} as Record<string, string[]>)
  }, [skus])

  // 查找匹配的SKU
  const selectedSku = useMemo(() => {
    // 如果没有选择任何属性，返回 null
    if (Object.keys(selectedAttributes).length === 0) return null

    // 检查是否所有属性都已选择
    const allAttributesSelected = Object.keys(attributes).every(
      key => selectedAttributes[key]
    )
    if (!allAttributesSelected) return null

    // 查找完全匹配的 SKU
    return skus.find(sku => {
      if (!sku.attributes) return false
      return Object.keys(selectedAttributes).every(
        key => sku.attributes![key] === selectedAttributes[key]
      )
    }) || null
  }, [skus, selectedAttributes, attributes])

  // 判断属性是否可选
  const isAttributeAvailable = (key: string, value: string) => {
    const tempAttrs = { ...selectedAttributes, [key]: value }

    return skus.some(sku => {
      if (!sku.attributes || sku.stock <= 0) return false

      // 检查临时属性是否匹配（只检查已选择的属性）
      return Object.keys(tempAttrs).every(
        k => sku.attributes![k] === tempAttrs[k]
      )
    })
  }

  // 选择属性
  const selectAttribute = (key: string, value: string) => {
    setSelectedAttributes(prev => ({ ...prev, [key]: value }))
  }

  // 重置选择
  const reset = () => {
    setSelectedAttributes({})
  }

  return {
    attributes,
    selectedAttributes,
    selectedSku,
    isAttributeAvailable,
    selectAttribute,
    reset,
  }
}
```

```typescript
import type { ProductSKU } from '@/types/product'

/**
 * 获取显示价格
 */
export function getDisplayPrice(
  selectedSku: ProductSKU | null,
  skus: ProductSKU[]
): string {
  if (selectedSku) {
    return `¥${selectedSku.price.toFixed(2)}`
  }

  const prices = skus.map(sku => sku.price)
  const minPrice = Math.min(...prices)
  const maxPrice = Math.max(...prices)

  if (minPrice === maxPrice) {
    return `¥${minPrice.toFixed(2)}`
  }

  return `¥${minPrice.toFixed(2)} - ¥${maxPrice.toFixed(2)}`
}

/**
 * 获取库存信息
 */
export function getStockInfo(
  selectedSku: ProductSKU | null,
  skus: ProductSKU[]
): { stock: number; available: boolean } {
  if (selectedSku) {
    return {
      stock: selectedSku.stock,
      available: selectedSku.stock > 0
    }
  }

  const totalStock = skus.reduce((sum, sku) => sum + sku.stock, 0)
  return {
    stock: totalStock,
    available: totalStock > 0
  }
}
```

### 5. QuantitySelector（数量选择器）

#### 文件: `src/pages/products/components/QuantitySelector.tsx`（新建）

```typescript
import { Button, InputNumber, message } from 'antd'
import { MinusOutlined, PlusOutlined } from '@ant-design/icons'

interface QuantitySelectorProps {
  value: number
  max: number
  onChange: (value: number) => void
}

export default function QuantitySelector({ value, max, onChange }: QuantitySelectorProps) {
  const handleDecrease = () => {
    if (value <= 1) return
    onChange(value - 1)
  }

  const handleIncrease = () => {
    if (value >= max) {
      message.warning('已达库存上限')
      return
    }
    onChange(value + 1)
  }

  const handleChange = (val: number | null) => {
    if (!val || val < 1) {
      onChange(1)
      return
    }

    if (val > max) {
      message.warning('超过库存数量')
      onChange(max)
      return
    }

    onChange(val)
  }

  return (
    <div className="flex items-center gap-4">
      <div className="flex items-center gap-2">
        <span className="text-sm text-gray-600">数量:</span>
        <div className="flex items-center border border-gray-300 rounded">
          <Button
            type="text"
            size="small"
            icon={<MinusOutlined />}
            onClick={handleDecrease}
            disabled={value <= 1}
          />
          <InputNumber
            value={value}
            onChange={handleChange}
            min={1}
            max={max}
            controls={false}
            className="w-16 text-center border-0"
          />
          <Button
            type="text"
            size="small"
            icon={<PlusOutlined />}
            onClick={handleIncrease}
            disabled={value >= max}
          />
        </div>
      </div>
      <span className="text-sm text-gray-500">库存: {max}件</span>
    </div>
  )
}
```

```typescript
import type { ProductSKU } from '@/types/product'

interface SKUSelectorProps {
  attributes: Record<string, string[]>
  selectedAttributes: Record<string, string>
  onAttributeChange: (key: string, value: string) => void
  isAttributeAvailable: (key: string, value: string) => boolean
  selectedSku: ProductSKU | null
}

export default function SKUSelector({
  attributes,
  selectedAttributes,
  onAttributeChange,
  isAttributeAvailable,
  selectedSku,
}: SKUSelectorProps) {
  // 属性名称映射（中文显示）
  const attrNameMap: Record<string, string> = {
    color: '颜色',
    size: '尺寸',
    style: '款式',
  }

  return (
    <div className="space-y-4 py-4 border-t border-b border-gray-200">
      {Object.entries(attributes).map(([key, values]) => (
        <div key={key}>
          <div className="text-sm text-gray-600 mb-2">
            {attrNameMap[key] || key}
          </div>
          <div className="flex flex-wrap gap-2">
            {values.map(value => {
              const isSelected = selectedAttributes[key] === value
              const isAvailable = isAttributeAvailable(key, value)

              return (
                <button
                  key={value}
                  onClick={() => onAttributeChange(key, value)}
                  disabled={!isAvailable}
                  className={`
                    px-4 py-2 border rounded transition-colors
                    ${isSelected
                      ? 'border-red-500 text-red-500 bg-red-50'
                      : 'border-gray-300 text-gray-700'
                    }
                    ${!isAvailable
                      ? 'opacity-50 cursor-not-allowed line-through'
                      : 'hover:border-red-500 cursor-pointer'
                    }
                  `}
                >
                  {value}
                </button>
              )
            })}
          </div>
        </div>
      ))}

      {/* 已选提示 */}
      {selectedSku && selectedSku.sku_name && (
        <div className="text-sm text-gray-500">
          已选: {selectedSku.sku_name}
        </div>
      )}
    </div>
  )
}
```

### 11. ProductDetailPage（主页面）

#### 文件: `src/pages/products/ProductDetailPage.tsx`（新建）

```typescript
import { useState } from 'react'
import { useParams, useNavigate } from 'react-router-dom'
import { Card, Spin, Alert, message } from 'antd'
import { useProductDetail } from '@/hooks/serverData/useProductDetail'
import { useSKUSelector } from '@/hooks/useSKUSelector'
import { getDisplayPrice, getStockInfo } from '@/utils/product'
import { cartApi } from '@/api/cart'
import ImageGallery from './components/ImageGallery'
import ProductInfo from './components/ProductInfo'
import SKUSelector from './components/SKUSelector'
import QuantitySelector from './components/QuantitySelector'
import ActionButtons from './components/ActionButtons'
import MerchantCard from './components/MerchantCard'
import ProductDetailTabs from './components/ProductDetailTabs'

export default function ProductDetailPage() {
  const { id } = useParams<{ id: string }>()
  const navigate = useNavigate()
  const [quantity, setQuantity] = useState(1)
  const [addingToCart, setAddingToCart] = useState(false)

  // 获取商品数据
  const { product, isLoading, error } = useProductDetail(Number(id))

  // SKU 选择逻辑
  const {
    attributes,
    selectedAttributes,
    selectedSku,
    isAttributeAvailable,
    selectAttribute,
  } = useSKUSelector(product?.skus || [])

  // 加载状态
  if (isLoading) {
    return (
      <div className="flex justify-center items-center min-h-screen">
        <Spin size="large" />
      </div>
    )
  }

  // 错误状态
  if (error || !product) {
    return (
      <div className="container mx-auto px-4 py-6">
        <Alert
          message="加载失败"
          description="商品信息加载失败，请稍后重试"
          type="error"
          showIcon
        />
      </div>
    )
  }

  // 计算价格和库存
  const displayPrice = getDisplayPrice(selectedSku, product.skus)
  const { stock, available } = getStockInfo(selectedSku, product.skus)

  // 加入购物车
  const handleAddToCart = async () => {
    if (!selectedSku) {
      message.warning('请选择商品规格')
      return
    }

    if (selectedSku.stock < quantity) {
      message.error('库存不足')
      return
    }

    try {
      setAddingToCart(true)
      await cartApi.addCart({
        sku_id: selectedSku.id,
        quantity: quantity,
      })
      message.success('已加入购物车')
    } catch (error) {
      message.error('加入购物车失败')
    } finally {
      setAddingToCart(false)
    }
  }

  // 立即购买
  const handleBuyNow = () => {
    if (!selectedSku) {
      message.warning('请选择商品规格')
      return
    }

    if (selectedSku.stock < quantity) {
      message.error('库存不足')
      return
    }

    navigate('/checkout', {
      state: {
        items: [{
          sku_id: selectedSku.id,
          quantity: quantity,
        }],
      },
    })
  }

  return (
    <div className="container mx-auto px-4 py-6">
      {/* 主要内容区 */}
      <Card className="mb-6">
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
          {/* 左侧：图片轮播 */}
          <ImageGallery images={product.images} video={product.video} />

          {/* 右侧：商品信息 */}
          <div>
            <ProductInfo
              name={product.name}
              price={displayPrice}
              sales={product.sales}
              categoryName={product.category_name}
            />

            <SKUSelector
              attributes={attributes}
              selectedAttributes={selectedAttributes}
              onAttributeChange={selectAttribute}
              isAttributeAvailable={isAttributeAvailable}
              selectedSku={selectedSku}
            />

            <QuantitySelector
              value={quantity}
              max={stock}
              onChange={setQuantity}
            />

            <ActionButtons
              onAddToCart={handleAddToCart}
              onBuyNow={handleBuyNow}
              loading={addingToCart}
              disabled={!available || !selectedSku}
            />
          </div>
        </div>
      </Card>

      {/* 商家信息 */}
      <MerchantCard
        merchantId={product.merchant_id}
        merchantName={product.merchant_name}
      />

      {/* 商品详情标签页 */}
      <Card>
        <ProductDetailTabs
          description={product.description}
          images={product.images}
        />
      </Card>
    </div>
  )
}
```

## 开发建议

1. **按顺序开发**: 严格按照第一阶段到第五阶段的顺序进行，每完成一个阶段再进入下一个
2. **单元测试**: 每完成一个组件就测试一次，确保功能正常
3. **SKU 逻辑**: SKU 选择器是最复杂的部分，建议先在独立文件中测试逻辑
4. **边界情况**: 注意处理单 SKU、无库存、商品下架等边界情况
5. **用户体验**: 所有操作都要有明确的反馈（Toast 提示、loading 状态等）

## 常见问题

### Q1: SKU 选择器如何处理单 SKU 商品？
A: 如果只有一个 SKU，可以在页面加载时自动选中，或者隐藏 SKU 选择器直接显示价格和库存。

### Q2: 如何处理无库存商品？
A: 禁用"加入购物车"和"立即购买"按钮，显示"暂时缺货"提示。

### Q3: 图片加载失败怎么办？
A: 使用 Ant Design 的 Image 组件，它自带加载失败的占位图。

### Q4: 如何优化性能？
A: 使用 useMemo 缓存 SKU 匹配逻辑，图片使用懒加载，路由使用 lazy 加载。
